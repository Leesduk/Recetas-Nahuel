    function renderMenu() {
      const grid = document.getElementById('menuGrid');
      grid.innerHTML = '';
      Object.entries(recipes).forEach(([key, recipe]) => {
        const card = document.createElement('div');
        card.className = 'fade-in bg-white/5 border border-white/10 rounded-2xl p-6 cursor-pointer hover:border-amber-400/50 hover:bg-white/10 transition';
        card.onclick = () => selectRecipe(key);
        card.innerHTML = `
          <div class="text-5xl mb-3">${recipe.emoji}</div>
          <h2 class="text-white font-bold text-xl mb-2">${recipe.name}</h2>
          <p class="text-slate-400 text-sm">${recipe.baseServings} raciones base</p>
        `;
        grid.appendChild(card);
      });
    }

    function selectRecipe(recipeKey) {
      currentRecipe = recipes[recipeKey];
      servings = currentRecipe.baseServings;
      document.getElementById('servingCount').textContent = servings;
      document.getElementById('title').textContent = currentRecipe.name;
      document.getElementById('menuScreen').classList.add('hidden');
      document.getElementById('recipeScreen').classList.remove('hidden');
      renderList();
      lucide.createIcons();
    }

    function backToMenu() {
      document.getElementById('recipeScreen').classList.add('hidden');
      document.getElementById('menuScreen').classList.remove('hidden');
      currentRecipe = null;
    }

    function getScaled(qty) {
      if (qty === 0) return "";
      const scaled = (qty / currentRecipe.baseServings) * servings;
      return scaled % 1 === 0 ? scaled : scaled.toFixed(1);
    }
